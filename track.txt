    src/
 ├── domain/           # Entidades e regras de negócio puras
 ├── application/      # Casos de uso e serviços
 ├── infrastructure/   # Implementações (DB, Redis, Pub/Sub)
 ├── interfaces/       # Controllers e DTOs (o "mundo exterior")
 ├── app.module.ts     # O "Root Module" que orquestra as camadas
 └── main.ts           # O ponto de entrada da aplicação (Bootstrap)
